    const Eigen::Matrix<double, 3, 1> & v̄,

    const Eigen::Matrix<double, 3, 1> & V̄,

def demo23(v̄, ō, ū, V̄, C_combining_macron_ₐ, θ, v, D_A, δ):
    """
    :param :v̄ : ℝ^3
    :param :ō : ℝ^3
    :param :ū : ℝ^3
    :param :V̄ : ℝ^3
    :param :C_combining_macron_ₐ : ℝ^3
    :param :θ : ℝ
    :param :v : ℝ
    :param :D_A : ℝ,ℝ->ℝ^3
