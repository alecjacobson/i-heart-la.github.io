/**
 * demo22
 *
 * @param U  ℝ^(3×3)
 * @param V  ℝ^(3×3)
 * @return T₁
 */
Eigen::Matrix<double, 3, 3> demo22(
    const Eigen::Matrix<double, 3, 3> & U,
    const Eigen::Matrix<double, 3, 3> & V)
{
    Eigen::Matrix<double, 3, 3> _T₁_0;
    _T₁_0 << 0, 0, 0,
    0, 0, -1,
    0, 1, 0;
    Eigen::Matrix<double, 3, 3> T₁ = 1 / double(sqrt(2)) * U * _T₁_0 * V.transpose();

    return T₁;
}

def demo22(U, V):
    """
    :param :U : ℝ^(3×3)
    :param :V : ℝ^(3×3)
    """
    U = np.asarray(U, dtype=np.float64)
    V = np.asarray(V, dtype=np.float64)

    assert U.shape == (3, 3)
    assert V.shape == (3, 3)

    _T1_0 = np.zeros((3, 3))
    _T1_0[0] = [0, 0, 0]
    _T1_0[1] = [0, 0, -1]

    T1 = 1 / np.sqrt(2) * U @ _T1_0 @ V.T
