    const std::vector<Eigen::Matrix<double, 3, 3>> & T,
    const std::vector<double> & α,
    const std::set<std::tuple< int > > & N₁_left_parenthesis_v_right_parenthesis,

            _sum_0 += α.at(i-1) * n(T.at(i-1));
        }
