double demo4(
    const std::vector<Eigen::Matrix<double, 3, 1>> & v,
    const std::vector<Eigen::Matrix<double, 3, 1>> & u,
    const std::vector<Eigen::Matrix<double, 2, 3>> & M,
    const std::vector<double> & A)
{
    const long _dim_0 = A.size();

def demo4(v, u, M, A):
    """
    :param :v : ℝ^3
    :param :u : ℝ^3
    :param :M : ℝ^(2×3)
    :param :A : ℝ
    """
    v = np.asarray(v, dtype=np.float64)
    u = np.asarray(u, dtype=np.float64)
    M = np.asarray(M, dtype=np.float64)
    A = np.asarray(A)

    _dim_0 = A.shape[0]
    assert v.shape == (_dim_0, 3, )
    assert u.shape == (_dim_0, 3, )
    assert M.shape == (_dim_0, 2, 3)
    assert A.shape == (_dim_0,)

    _sum_0 = 0
    for T in range(1, len(M)+1):
        _n_0 = np.zeros((2, 2))
        _n_0[0] = [0, -1]
        _n_0[1] = [1, 0]
        _sum_0 += A[T-1] * np.power(np.linalg.norm(M[T-1] @ v[T-1] - _n_0 @ M[T-1] @ u[T-1], 2), 2)
    n = _sum_0

    return n
